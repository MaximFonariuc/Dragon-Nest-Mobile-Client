--主动派发注册消息
-- PTC
function LuaNotifyProcess.Process(itype,data,length)  
	--print("process: "..type)
    
    local find = false

    for k,v in pairs(PTC) do
    	if k == itype then
    		local func = v[1]
    		if func ~= nil then
    		   func(data,length)
    		end
    		find = true
    		break
    	end
    end

    if not find then
    	for k,v in pairs(PtcCB) do
    		if v == itype then
    			local func = v[1]
    			if func ~= nil then
    				func(data,length)
    			end
    			break
    		end
    	end
    end
end


-- NetworkOutCSharp屏蔽的网络协议
-- 只有回调通知
function LuaNotifyProcess.ProcessOveride(itype,data,length)
	for k,v in pairs(NetworkOverideCSharp) do
		if k == itype then
			local func = v[1]
			if func ~= nil then
				func(data,length)
			end
			break
		end
	end
end


--处理注册消息ID
function LuaNotifyProcess.FetchRegistedID()
	local registeds = {} 

	local index = 1
	for k,v in pairs(PTC) do
		--print("k: "..tostring(k).." v: "..tostring(v[2]))
		registeds[index] = k
		registeds[index+1] = 1  --copybuff
		index = index + 2
	end

	for k,v in pairs(PtcCB) do
		registeds[index]= k
		registeds[index+1] = 0  --nocppy
		index = index + 2
	end

	for k,v in pairs(NetworkOverideCSharp) do
		registeds[index] = k
		registeds[index + 1] = 2 --not dispacher in c#
		index = index + 2
	end

	return unpack(registeds)
end
